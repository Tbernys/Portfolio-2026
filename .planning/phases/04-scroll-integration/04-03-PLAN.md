// Back-to-top button — appears after hero scroll-out
const backToTopBtn = document.getElementById('back-to-top');

// Show after hero fades (progress > 0.95 means hero is gone)
// Use a separate ScrollTrigger watching the same trigger as Track B
ScrollTrigger.create({
  trigger: '#hero',
  start: 'top top',

      backToTopBtn.classList.add('is-visible');
    } else {
      backToTopBtn.classList.remove('is-visible');
    }
  }
});

backToTopBtn.addEventListener('click', () => {

});
