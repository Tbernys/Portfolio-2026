function initProjectsShowcase() {
  const grid = document.getElementById('projects-grid');
  if (!grid) return;

  // Generate card HTML for all 9 projects (synchronous — DOM first, thumbnails async)
  grid.innerHTML = PROJECTS.map((proj, i) => `
    <div class="project-card" data-vimeo-id="${proj.id}" data-index="${i}" tabindex="0" role="button" aria-label="${proj.title}">
      <img class="project-thumb" src="" alt="${proj.title}" loading="lazy">
      <div class="project-overlay">
        <div class="project-overlay-title">${proj.title}</div>
        <div class="project-overlay-role">${proj.role}</div>
      </div>
    </div>
  `).join('');

  // Fetch thumbnails via oEmbed in parallel (non-blocking, graceful fallback)

  // Set up IntersectionObserver for lazy-load readiness

}

**3. Add `fetchThumbnails()` function:**

```javascript
async function fetchThumbnails() {
  const requests = PROJECTS.map(p =>
    fetch(`https://vimeo.com/api/oembed.json?url=https://vimeo.com/${p.id}&width=640`)
      .then(r => r.json())
      .then(data => ({ id: p.id, url: data.thumbnail_url }))
      .catch(() => ({ id: p.id, url: null }))
  );
  const results = await Promise.all(requests);
  results.forEach(({ id, url }) => {
    if (!url) return;
    // Set src on ALL cards with this vimeo ID (handles duplicates in placeholder data)
    document.querySelectorAll(`[data-vimeo-id="${id}"] .project-thumb`).forEach(img => {
      img.src = url;
    });
  });
}
```

**4. Add `setupCardObserver()` function:**

```javascript
function setupCardObserver() {
  const cardObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.dataset.nearViewport = 'true';
        cardObserver.unobserve(entry.target);
      }
    });
  }, { rootMargin: '200px' });

  document.querySelectorAll('.project-card').forEach(card => cardObserver.observe(card));
}
```
